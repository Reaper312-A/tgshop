@dataclass
class Product:
    id: int
    name: str

    subcategory: str
    price: float
    currency: str

    short_description: str
    quantity: int
    available: bool = True
    
    @property
    def photo_path(self):
        """–ü—É—Ç—å –∫ —Ñ–æ—Ç–æ —Ç–æ–≤–∞—Ä–∞"""
        from config import PRODUCT_PHOTOS_DIR, DEFAULT_PRODUCT_PHOTO
        # –§–æ—Ä–º–∏—Ä—É–µ–º –∏–º—è —Ñ–∞–π–ª–∞: –∫–∞—Ç–µ–≥–æ—Ä–∏—è_–ø–æ–¥–∫–∞—Ç–µ–≥–æ—Ä–∏—è_id.jpg

    def category_photo_path(self):
        """–ü—É—Ç—å –∫ —Ñ–æ—Ç–æ –∫–∞—Ç–µ–≥–æ—Ä–∏–∏"""
